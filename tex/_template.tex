    \tikz[overlay]\node[%
        fill=blue!20,%
        fill opacity=1.0,%
        text opacity=1,%
        inner sep=1.5pt,%
        anchor=text,%
        rectangle,%
        rounded corners=0.75mm,%
        text depth=0pt,%
        inner xsep=0pt,%
        #1]%
    {#2};%
    \phantom{#2}%
}
